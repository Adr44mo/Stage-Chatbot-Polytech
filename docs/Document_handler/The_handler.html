145@router.post("/add_site")
146def add_site(data: AddSiteInput):
147    try:
148        # Vérification que l'URL commence par "https://"
149        if not data.url.startswith("https://"):
150            raise HTTPException(status_code=400, detail="L'URL doit commencer par 'https://'")
151        
152        # Récupération de tous les sites existants pour vérifier l'unicité
153        existing_sites = []
154        for f in CONFIG_DIR.glob("*.yaml"):
155            with open(f, "r", encoding="utf-8") as file:
156                site_data = yaml.safe_load(file)
157                existing_sites.append({
158                    "name": site_data.get("NAME"),
159                    "url": site_data.get("BASE_URL")
160                })
161        
162        # Vérification que le nom du site n'existe pas déjà
163        existing_names = [site["name"].lower() for site in existing_sites if site["name"]]
164        if data.siteName.lower() in existing_names:
165            raise HTTPException(status_code=400, detail=f"Un site avec le nom '{data.siteName}' existe déjà")
166        
167        # Vérification que l'URL n'existe pas déjà
168        existing_urls = [site["url"] for site in existing_sites if site["url"]]
169        if data.url in existing_urls:
170            raise HTTPException(status_code=400, detail=f"Un site avec l'URL '{data.url}' existe déjà")
171        
172        generate_config(data.siteName, data.url)
173        return {"status": "success", "message": f"Site {data.siteName} ajouté avec succès."}
174    except HTTPException:
175        # Re-raise les HTTPException pour qu'elles soient renvoyées avec le bon status code
176        raise
177    except Exception as e:
178        raise HTTPException(status_code=500, detail=f"Erreur lors de l'ajout du site : {e}")

154        existing_sites = []
155        for f in CONFIG_DIR.glob("*.yaml"):
156            with open(f, "r", encoding="utf-8") as file:
157                site_data = yaml.safe_load(file)
158                existing_sites.append({
159                    "name": site_data.get("NAME"),
160                    "url": site_data.get("BASE_URL")
161                })
162        
163        # Vérification que le nom du site n'existe pas déjà
164        existing_names = [site["name"].lower() for site in existing_sites if site["name"]]
165        if data.siteName.lower() in existing_names:
166            raise HTTPException(status_code=400, detail=f"Un site avec le nom '{data.siteName}' existe déjà")
167        
168        # Vérification que l'URL n'existe pas déjà
169        existing_urls = [site["url"] for site in existing_sites if site["url"]]
170        if data.url in existing_urls:
171            raise HTTPException(status_code=400, detail=f"Un site avec l'URL '{data.url}' existe déjà")
172        
173        generate_config(data.siteName, data.url)
174        return {"status": "success", "message": f"Site {data.siteName} ajouté avec succès."}
175    except HTTPException:
176        # Re-raise les HTTPException pour qu'elles soient renvoyées avec le bon status code
