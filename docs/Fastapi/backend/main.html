164    session_id = str(uuid.uuid4())

167    cp.print_debug(f"[init-session] Création nouvelle session: {session_id}")

170        key="polybot_session_id",
171        value=session_id,
172        httponly=True,

176        path="/"
177    )
178    cp.print_debug(f"[init-session] Cookie défini pour nouvelle session: {session_id}")
179    return {"ok": True, "session_id": session_id}

187@limiter.limit("10/second; 60/minute; 500/hour")  # Limite par IP (clé par défaut)

189async def chat(request: Request, request_body: ChatRequest, polybot_session_id: str = Cookie(None), session: Session = Depends(get_session)):
190    """
191    Endpoint principal pour le chatbot RAG : gestion des messages et historique de conversation.
192    """
193    try:
194        # Debug des cookies reçus
195        cookies_debug = request.headers.get("cookie", "Aucun cookie")
196        cp.print_debug(f"[chat] Cookies reçus: {cookies_debug}")
197        cp.print_debug(f"[chat] polybot_session_id extrait: {polybot_session_id}")
198        
199        # Vérification reCAPTCHA
200        recaptcha_validated = request.headers.get("X-Recaptcha-Validated")
201        if not request_body.recaptcha_token:
202            # Si le frontend indique que le captcha a déjà été validé, on accepte
203            if recaptcha_validated == "true":
204                pass
205            else:
206                raise HTTPException(status_code=400, detail="reCAPTCHA token required")
207        elif not await verify_recaptcha_token(request_body.recaptcha_token):
208            raise HTTPException(status_code=400, detail="Invalid reCAPTCHA token")
209
210        cp.print_debug(f"User prompt: {request_body.prompt}")
211
212        conversation = get_or_create_conversation(session, polybot_session_id)
213        add_message(session, conversation.id, "user", request_body.prompt)
214
215        # ====================================================
216        # TEST : Intelligent RAG vs LangGraph vs RAG classique
217        # ====================================================
218
219        if USE_INTELLIGENT_RAG:
220            chat_history = [
221                {"role": msg.role, "content": msg.content}
222                for msg in request_body.chat_history if msg.role == "assistant" or msg.role == "user"
223            ]
224            try:
225                response = invoke_intelligent_rag(request_body.prompt, chat_history, False)
226            except Exception as e:
227                cp.print_error(f"Error in invoke_intelligent_rag: {e}")
228                raise HTTPException(status_code=500, detail=f"Error in intelligent RAG: {str(e)}")
229            answer = response.get("answer", "")
230            context = response.get("context", [])
231            sources = response.get("sources", [])
232        elif USE_LANGGRAPH:
233            try:
234                response = invoke_langgraph_rag({
235                    "input": request_body.prompt,
236                    "chat_history": [
237                        {"role": msg.role, "content": msg.content}
238                        for msg in request_body.chat_history if msg.role == "assistant" or msg.role == "user"
239                    ],
240                })
241            except Exception as e:
242                cp.print_error(f"Error in invoke_langgraph_rag: {e}")
243                raise HTTPException(status_code=500, detail=f"Error in LangGraph RAG: {str(e)}")
244            answer = response.get("answer", "")
245            context = response.get("context", [])
246            sources = get_sources(context) if context else []
247        else:
248            try:
249                response = rag_chain.invoke({
250                    "input": request_body.prompt,
251                    "chat_history": [
252                        {"role": msg.role, "content": msg.content}
253                        for msg in request_body.chat_history if msg.role == "assistant" or msg.role == "user"
254                    ],
255                })
256            except Exception as e:
257                cp.print_error(f"Error in rag_chain.invoke: {e}")
258                raise HTTPException(status_code=500, detail=f"Error in classic RAG: {str(e)}")
259            answer = response.get("answer", "")
260            context = response.get("context", [])
261            sources = get_sources(context) if context else []
262
263        try:
264            add_message(session, conversation.id, "assistant", answer, sources)
265            update_rag_conversation(
266                invoke_result=response,
267                conversation=conversation,
268                session=session
269            )
270            
271        except Exception as e:
272            cp.print_error(f"Error in add_message: {e}")
273            raise HTTPException(status_code=500, detail=f"Error saving assistant message: {str(e)}")
274
275        return ChatResponse(answer=answer, sources=sources)
276    except HTTPException as http_exc:
277        raise http_exc
278    except Exception as e:
279        cp.print_error(f"Unexpected error in /chat endpoint: {e}")
280        raise HTTPException(status_code=500, detail=f"Unexpected error: {str(e)}")

280        cp.print_error(f"Unexpected error in /chat endpoint: {e}")
